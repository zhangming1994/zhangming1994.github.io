得到文件信息

package main

import (
    "fmt"
    "log"
    "os"
)

func main() {
    //文件不存在会报错
    fileInfo, err := os.Stat("test.txt")
    if err != nil {
        log.Fatal(err)
    }
    fmt.Println("file name:", fileInfo.Name())       //文件名称
    fmt.Println("size", fileInfo.Size())             //文件大小
    fmt.Println("permissions", fileInfo.Mode())      //权限
    fmt.Println("last modified", fileInfo.ModTime()) //最后修改时间
    fmt.Println("is directory", fileInfo.IsDir())    //是不是目录
    fmt.Println("type:%T\n", fileInfo.Sys())
    fmt.Println("sysinfo:%+v\n\n", fileInfo.Sys())
}

打开和关闭文件

package main

import (
    "log"
    "os"
)

func main() {
    file, err := os.Open("分布式.txt")
    if err != nil {
        log.Fatal(err)
    }
    file.Close()
    //os.OpenFile提供更多选项(1.权限模式2.打开时候的属性)
    file, err = os.OpenFile("分布式.txt", os.O_APPEND, 0666)
    if err != nil {
        log.Fatal(err)
    }
    file.Close()
    //文件权限 1.执行 2.写入 4.读取
    //os.O_RDONLY 只读
    //os.O_WRONLU 只写
    //os.O_RDWR 读写
    //os.O_APPEND 往文件中间添加
    //os.O_CREATE 文件不存在先创建文件
    //os.O_TRUNC 文件打开的时候裁剪文件
    //os.O_EXCL 和create一起使用，文件不能存在
    //os.O_SYNC 以同步i/o方式打开
}

复制文件

package main

import (
    "io"
    "log"
    "os"
)

func main() {
    //打开文件
    file, err := os.Open("test.txt")
    if err != nil {
        log.Fatal(err)
    }
    defer file.Close()
    newFile, err := os.Create("testnew.txt")
    if err != nil {
        log.Fatal(err)
    }
    defer newFile.Close()
    //文件复制
    _, err = io.Copy(newFile, file)
    if err != nil {
        log.Fatal(err)
    }
    //flush内容
    err = newFile.Sync()
    if err != nil {
        log.Fatal(err)
    }
}

跳转到文件制定位置(seek)

package main

import (
    "fmt"
    "log"
    "os"
)

func main() {
    file, _ := os.Open("test.txt")
    defer file.Close()

    var offset int64 = 5 //偏离位置
    var whence int = 0   //初始位置(0.文件开始位置1.当前位置2.文件结尾位置)

    newposition, err := file.Seek(offset, whence)
    if err != nil {
        log.Fatal(err)
    }
    fmt.Println(newposition)
    //回退
    newposition, err = file.Seek(-2, 1)
    if err != nil {
        log.Fatal(err)
    }
    fmt.Println(newposition)
    //得到当前位置
    currentPosition, err := file.Seek(0, 1)
    fmt.Println(currentPosition)
    //跳转到文件开始的地方
    newposition, err = file.Seek(0, 0)
    if err != nil {
        log.Fatal(err)
    }
    fmt.Println(newposition)
}

####使用缓存写文件
package main

import (
    "bufio"
    "io/ioutil"
    "log"
    "os"
)

//使用os包写入一个打开的文件，
//go的可执行包是静态链接的可执行文件,import的每一个包都会增加可执行文件的大小
func main() {
    file, err := os.OpenFile("test.txt", os.O_WRONLY|os.O_TRUNC|os.O_CREATE, 0666)
    if err != nil {
        log.Fatal(err)
    }
    defer file.Close()
    //写字节到文件中
    byteSlice := []byte("Bytes!\n")
    bytesWritten, err := file.Write(byteSlice)
    if err != nil {
        log.Fatal(err)
    }
    log.Println(bytesWritten)
    WriteSlice()
}

//ioutil有一个方法WriteFile()可以处理创建/打开文件,写字节slice和关闭文件等一系列操作
//适用于简介快速的写字节slice到文件中
func WriteSlice() {
    err := ioutil.WriteFile("test.txt", []byte("kasdjfkjsaksjdfkljlskdf\n"), 0666)
    if err != nil {
        log.Fatal(err)
    }
}

//使用缓存写
//bufio提供了带缓存的writer,这样可以在写字节到硬盘前使用内存缓存，档处理很多数据的时候有用
//节省了操作硬盘io的时间，减少硬盘的磨损提升性能
func BufioWrite() {
    file, err := os.OpenFile("test.txt", os.O_WRONLY, 0666)
    if err != nil {
        log.Fatal(err)
    }
    defer file.Close()
    //创建bufio
    bfuueredWriter := bufio.NewWriter(file)
    //写字节到buffer
    bytesWriten, err := bfuueredWriter.Write([]byte{65, 68, 89})
    if err != nil {
        log.Fatal(err)
    }
    log.Println(bytesWriten)
    //写字符串到buffer
    //也可以使用WriteRune(),WriteByte()
    bytesWriten, err = bfuueredWriter.WriteString("bufferfed string\n")
    if err != nil {
        log.Fatal(err)
    }
    //检查缓存中间的字节数
    unflushbufferSize := bfuueredWriter.Buffered()
    //还有多少字节可以用
    bytesAvailable := bfuueredWriter.Available()
    //写入硬盘
    bfuueredWriter.Flush()

    //扩容
    //第一个蚕食是缓存输出到哪里，设置的新的大小小于第一个参数writer的缓存大小，会得到writer的原始大小的缓存，默认是4096
    bufferedWriter = bufio.NewWriterSize(bfuueredWriter, 8000)
    //再次检查缓存大小
    bytesAvailable = bfuueredWriter.Available()
    if err != nil {
        log.Fatal(err)
    }
}

####使用缓存读取
package main

import (
    "bufio"
    "io"
    "io/ioutil"
    "log"
    "os"
)

func main() {
    //最多读取N个字节
    file, err := os.Open("test.txt")
    if err != nil {
        log.Fatal(err)
    }
    //返回0字节意味读取到文件末尾了，会返回io.EOF的错误
    defer file.Close()
    byteSlice := make([]byte, 16)
    bytesRead, err := file.Read(byteSlice)
    if err != nil {
        log.Fatal(err)
    }
    log.Println(bytesRead, byteSlice)
    //正好读取N个字节
    ReadNbytes()
    //读取至少N个字节
    ReadLastN()
    //读取全部字节
    ReadAll()
    //快速读取到内存
    ReadTo()
    //使用缓存读取
    ReadBuffio()
}
func ReadNbytes() {
    file, err := os.Open("test.txt")
    if err != nil {
        log.Fatal(err)
    }
    //file.Read()可以读取一个小文件到大的byte slice中，
    //io.ReadFull()在文件的字节数小于byte slice字节数的时候会返回错误
    bytesSlice := make([]byte, 2)
    numBytesRead, err := io.ReadFull(file, bytesSlice)
    if err != nil {
        log.Fatal(err)
    }
    log.Println(numBytesRead, bytesSlice)
}

func ReadLastN() {
    file, err := os.Open("test.txt")
    if err != nil {
        log.Fatal(err)
    }
    bytesSlice := make([]byte, 512)
    minBytes := 8
    //io.readatleast在不能得到最小的字节的时候会返回错误，但会吧已经读的文件保留
    numBytesRead, err := io.ReadAtLeast(file, bytesSlice, minBytes)
    if err != nil {
        log.Fatal(err)
    }
}

func ReadAll() {
    file, err := os.Open("test.txt")
    if err != nil {
        log.Fatal(err)
    }
    //os.File.Read(),io.ReadFull(),io.ReadAtLeast()在读取之前都需要一个固定大小的byte slice
    //但是ioutil.ReadAll不需要,他会读取reader的每一个字节然后返回slice
    data, err := ioutil.ReadAll(file)
    if err != nil {
        log.Fatal(err)
    }
}

func ReadTo() {
    // 读取文件到byte slice中
    data, err := ioutil.ReadFile("test.txt")
    if err != nil {
        log.Fatal(err)
    }
    log.Printf("Data read: %s\n", data)
}
func ReadBuffio() {
    file, err := os.Open("test.txt")
    if err != nil {
        log.Fatal(err)
    }
    bufferedReader := bufio.NewReader(file)
    byteSlice := make([]byte, 5)
    byteSlice, err = bufferedReader.Peek(5)
    if err != nil {
        log.Fatal(err)
    }
    numBytesRead, err := bufferedReader.Read(byteSlice)
    if err != nil {
        log.Fatal(err)
    }
    //读取一个字节
    myBytes, err := bufferedReader.ReadByte()
    if err != nil {
        log.Fatal(err)
    }
    //读取到分隔符就返回slice
    dataBytes, err := bufferedReader.ReadBytes('\n')
    if err != nil {
        log.Fatal(err)
    }
    //读取返回字符串
    dataString, err := bufferedReader.ReadString('\n')
    if err != nil {
        log.Fatal(err)
    }
}

压缩文件

package main

import (
    "archive/zip"
    "log"
    "os"
)

func main() {
    //创建一个打包文件
    outFile, err := os.Create("test.zip")
    if err != nil {
        log.Fatal(err)
    }
    defer outFile.Close()
    //创建zip writer
    zipWritter := zip.NewWriter(outFile)
    //向打包文件中间写文件
    var filesToArchive = []struct {
        Name, Body string
    }{
        {"test.txt", "String contents of file"},
        {"test2.txt", "\x61\x62\x63\n"},
    }
    //写入
    for _, file := range filesToArchive {
        //创建
        fileWriter, err := zipWritter.Create(file.Name)
        if err != nil {
            log.Fatal(err)
        }
        _, err = fileWriter.Write([]byte(file.Body))
        if err != nil {
            log.Fatal(err)
        }
    }
    err = zipWritter.Close()
    if err != nil {
        log.Fatal(err)
    }
}

解压缩

package main

import (
    "archive/zip"
    "io"
    "log"
    "os"
    "path/filepath"
)

func main() {
    zipReader, err := zip.OpenReader("test.zip")
    if err != nil {
        log.Fatal(err)
    }
    defer zipReader.Close()
    //遍历
    for _, file := range zipReader.Reader.File {
        zippedFile, err := file.Open()
        if err != nil {
            log.Fatal(err)
        }
        defer zippedFile.Close()
        //指定抽取到的位置
        targetDir := "./"
        extractedFilePath := filepath.Join(
            targetDir, //文件位置
            file.Name, //文件名称
        )
        if file.FileInfo().IsDir() { //是目录
            os.MkdirAll(extractedFilePath, file.Mode())
        } else { //是文件n
            outpurFile, err := os.OpenFile(extractedFilePath, os.O_WRONLY|os.O_CREATE|os.O_TRUNC, file.Mode())
            if err != nil {
                log.Fatal(err)
            }
            defer outpurFile.Close()
            //赋值文件
            _, err = io.Copy(outpurFile, zippedFile)
            if err != nil {
                log.Fatal(err)
            }
        }
    }
}

###gzip压缩解压缩
package main

import (
    "compress/gzip"
    "io"
    "log"
    "os"
)

func main() {
    outputFile, err := os.Create("test.txt.gz")
    if err != nil {
        log.Fatal(err)
    }
    gzipWriter := gzip.NewWriter(outputFile)
    defer gzipWriter.Close()
    _, err = gzipWriter.Write([]byte("阿克江收到回复会计核算\n"))
    if err != nil {
        log.Fatal(err)
    }
    //解压缩
    ungzip()
}

func ungzip() {
    gzipFile, err := os.Open("test.txt.gz")
    if err != nil {
        log.Fatal(err)
    }
    gzipReader, err := gzip.NewReader(gzipFile) //reder
    if err != nil {
        log.Fatal(err)
    }
    defer gzipReader.Close()
    outfileWritermerr := os.Create("unzipped.txt") //创建文件
    if err != nil {
        log.Fatal(err)
    }
    defer outfileWritermerr.Close()
    _, err = io.Copy(outfileWritermerr, gzipReader) //文件内容复制
    if err != nil {
        log.Fatal(err)
    }
}

get请求下载

package main

import (
    "io"
    "log"
    "net/http"
    "os"
)

func main() {
    newFile, err := os.Create("devdungeon.html")
    if err != nil {
        log.Fatal(err)
    }
    defer newFile.Close()
    URL := "http://www.devdungeon.com/archive"
    response, err := http.Get(URL)
    defer response.Body.Close()
    numbytes, err := io.Copy(newFile, response.Body)
    if err != nil {
        log.Fatal(err)
    }
    log.Println(numbytes)
}
