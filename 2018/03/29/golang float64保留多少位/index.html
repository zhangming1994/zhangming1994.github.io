第一种思路，利用利用fmt.Sprintf(),这种用到反射，效率不高

func Round2(f float64, n int) float64 {  
    floatStr := fmt.Sprintf("%."+strconv.Itoa(n)+"f", f)  
    inst, _ := strconv.ParseFloat(floatStr, 64)  
    return inst  
}

第二种思路，利用math包，golang的浮点数计算结果并不是很准确加上四舍五入的时候用的是银行舍入法，和其他语言的四舍五入值对不上

//先获取保留小数点后面位数的整数比，如两位小数就是100，
func Round(f float64, n int) float64 {
    pow10_n := math.Pow10(n)//对10取n次幂
    //然后获取一个小数位比需要的大一位的浮点数，加上传递进来的浮点数，然后乘以10的n次幂，得到一个小数点后一位的
    //浮点数，然后math.trunc取整,除以10的n次幂，去掉加权小数
    return math.Trunc((f+0.5/pow10_n)*pow10_n) / pow10_n
}

这里详细了解一下Pow10()和math.Rrunc()

这是Pow10()源码
func Pow10(n int) float64 {
if 0 <= n && n <= 308 {
    return pow10postab32[uint(n)/32] * pow10tab[uint(n)%32]
}

if -323 <= n && n <= 0 {
    return pow10negtab32[uint(-n)/32] / pow10tab[uint(-n)%32]
}

// n < -323 || 308 < n
if n > 0 {
    return Inf(1)
}

// n < -323
return 0
}


// pow10postab32 stores the pre-computed value for 10**(i*32) at index i.
var pow10postab32 = [...]float64{
    1e00, 1e32, 1e64, 1e96, 1e128, 1e160, 1e192, 1e224, 1e256, 1e288,
}

// pow10negtab32 stores the pre-computed value for 10**(-i*32) at index i.
var pow10negtab32 = [...]float64{
    1e-00, 1e-32, 1e-64, 1e-96, 1e-128, 1e-160, 1e-192, 1e-224, 1e-256, 1e-288, 1e-320,
}
